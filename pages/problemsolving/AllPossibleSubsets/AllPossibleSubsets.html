Find All Possible Subsets

This famous interview question can be solved using recursion. More specifically, all we need to do is to make a decision tree based on recursion. Let's take a look at this diagram:

Implementation
class SubsetElement { public: int val; bool isEmpty; SubsetElement() : isEmpty(true) {} SubsetElement(int i) : val(i), isEmpty(false) {} }; void printSubset(const vector& subset) { string str = "{"; for (size_t i = 0; i < subset.size(); i++) { if (!subset[i].isEmpty) { str += to_string(subset[i].val); str += ","; } } if (str.length() > 1) { str.pop_back(); } str += "}"; cout << str << endl; } void recurse(const vector& vec, vector subset, int at, int& subsetCount) { if (at == vec.size()) { subsetCount++; printSubset(subset); return; } SubsetElement emptyElement; subset.push_back(emptyElement); recurse(vec, subset, at+1, subsetCount); SubsetElement intElement(vec[at]); subset[subset.size()-1] = intElement; recurse(vec, subset, at+1, subsetCount); } void printAllSubsets(const vector& vec) { vector subset; int totalSubsets = 0; recurse(vec, subset, 0, totalSubsets); cout << "\nTotal subsets: " << totalSubsets << endl; } int main() { vector vec = {1, 2, 3, 4}; printAllSubsets(vec); return 0; }
